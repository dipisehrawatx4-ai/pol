<script>

function calculateScore() {


    let total = 0;


    let sections = {

        electoral: { score: 0, max: 60 },        // Q1–Q12

        government: { score: 0, max: 40 },       // Q13–Q20

        participation: { score: 0, max: 40 },    // Q21–Q28

        culture: { score: 0, max: 30 },          // Q29–Q34

        liberties: { score: 0, max: 70 }          // Q35–Q48

    };


    for (let i = 0; i < 48; i++) {

        const selected = document.querySelector(`input[name="q${i}"]:checked`);

        if (!selected) {

            alert("Please answer all questions before submitting.");

            return;

        }


        const value = parseInt(selected.value);

        total += value;


        if (i <= 11) sections.electoral.score += value;

        else if (i <= 19) sections.government.score += value;

        else if (i <= 27) sections.participation.score += value;

        else if (i <= 33) sections.culture.score += value;

        else sections.liberties.score += value;

    }


    const finalScore = (total / 24).toFixed(2);


    const electoral10 = (sections.electoral.score / sections.electoral.max * 10).toFixed(2);

    const government10 = (sections.government.score / sections.government.max * 10).toFixed(2);

    const participation10 = (sections.participation.score / sections.participation.max * 10).toFixed(2);

    const culture10 = (sections.culture.score / sections.culture.max * 10).toFixed(2);

    const liberties10 = (sections.liberties.score / sections.liberties.max * 10).toFixed(2);


    document.getElementById("result").innerHTML = `

        <h2>Your Results</h2>


        <p><strong>Overall Democracy Score:</strong> ${finalScore} / 10</p>

        <p><strong>Raw Score:</strong> ${total} / 240</p>


        <hr>


        <h3>Section-wise Scores</h3>

        <p>Electoral Process & Pluralism: ${sections.electoral.score} / 60 (${electoral10} / 10)</p>

        <p>Functioning of Government: ${sections.government.score} / 40 (${government10} / 10)</p>

        <p>Political Participation: ${sections.participation.score} / 40 (${participation10} / 10)</p>

        <p>Democratic Political Culture: ${sections.culture.score} / 30 (${culture10} / 10)</p>

        <p>Civil Liberties: ${sections.liberties.score} / 70 (${liberties10} / 10)</p>

    `;

}

</script>
